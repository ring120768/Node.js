/**
 * Dash-cam Upload Bucket Creation
 * 
 * Idempotent function to create the "dash-cam-upload" Supabase Storage bucket
 * if it doesn't already exist. This should be run server-side with admin privileges.
 * 
 * Add this to your main server file (index.js) or create a separate initialization script.
 */

/**
 * Create the dash-cam-upload bucket if it doesn't exist
 * @param {Object} supabase - Supabase admin client
 * @returns {Promise<boolean>} - Success status
 */
async function createDashcamBucketIfMissing(supabase) {
    const bucketName = 'dash-cam-upload';
    
    try {
        console.log('🗄️ Checking if dash-cam-upload bucket exists...');
        
        // Check if bucket already exists
        const { data: buckets, error: listError } = await supabase.storage.listBuckets();
        
        if (listError) {
            console.error('Failed to list storage buckets:', listError);
            return false;
        }
        
        // Check if our bucket exists
        const existingBucket = buckets.find(bucket => bucket.name === bucketName);
        
        if (existingBucket) {
            console.log('✅ dash-cam-upload bucket already exists');
            return true;
        }
        
        // Create the bucket
        console.log('📦 Creating dash-cam-upload bucket...');
        
        const { data, error } = await supabase.storage.createBucket(bucketName, {
            public: false, // Private bucket - access via signed URLs only
            fileSizeLimit: 1073741824, // 1GB limit
            allowedMimeTypes: [
                'video/mp4',
                'video/webm', 
                'video/quicktime',
                'video/x-msvideo',
                'video/avi',
                'video/mov'
            ]
        });
        
        if (error) {
            console.error('Failed to create dash-cam-upload bucket:', error);
            return false;
        }
        
        console.log('✅ dash-cam-upload bucket created successfully');
        
        // Set up RLS policies for the bucket
        await setupStorageRLSPolicies(supabase, bucketName);
        
        return true;
        
    } catch (error) {
        console.error('Error in createDashcamBucketIfMissing:', error);
        return false;
    }
}

/**
 * Initialize dash-cam upload functionality
 * Call this during your application startup
 */
async function initializeDashcamUpload() {
    // Only run if Supabase is configured
    if (!supabaseEnabled || !supabase) {
        console.warn('⚠️  Supabase not configured - dash-cam upload bucket creation skipped');
        return false;
    }
    
    try {
        // Create bucket if missing
        const bucketCreated = await createDashcamBucketIfMissing(supabase);
        
        if (bucketCreated) {
            console.log('🎥 Dash-cam upload system initialized successfully');
            
            // Add endpoint to get signed URLs for video playback
            setupDashcamEndpoints();
            
            return true;
        } else {
            console.warn('⚠️  Failed to initialize dash-cam upload system');
            return false;
        }
        
    } catch (error) {
        console.error('Error initializing dash-cam upload:', error);
        return false;
    }
}

/**
 * Set up API endpoints for dash-cam functionality
 */
function setupDashcamEndpoints() {
    // Get signed URL for video playback
    app.get('/api/dashcam/signed-url/:userId/:incidentId/:filename', checkSharedKey, async (req, res) => {
        try {
            const { userId, incidentId, filename } = req.params;
            const expiresIn = parseInt(req.query.expires) || 3600; // Default 1 hour
            
            // Verify user owns this file by checking the database
            const { data: evidence, error } = await supabase
                .from('incident_evidence')
                .select('storage_path')
                .eq('user_id', userId)
                .eq('incident_id', incidentId)
                .eq('kind', 'video')
                .eq('storage_path', `${userId}/${incidentId}/${filename}`)
                .single();
            
            if (error || !evidence) {
                return res.status(404).json({
                    error: 'Video not found or access denied',
                    requestId: req.requestId
                });
            }
            
            // Generate signed URL
            const { data: urlData, error: urlError } = await supabase.storage
                .from('dash-cam-upload')
                .createSignedUrl(evidence.storage_path, expiresIn);
            
            if (urlError) {
                throw urlError;
            }
            
            res.json({
                success: true,
                signedUrl: urlData.signedUrl,
                expiresIn: expiresIn,
                expiresAt: new Date(Date.now() + (expiresIn * 1000)).toISOString(),
                requestId: req.requestId
            });
            
        } catch (error) {
            console.error('Error generating signed URL:', error);
            res.status(500).json({
                error: 'Failed to generate signed URL',
                details: error.message,
                requestId: req.requestId
            });
        }
    });
    
    // Get user's uploaded videos for an incident
    app.get('/api/dashcam/videos/:userId/:incidentId', checkSharedKey, async (req, res) => {
        try {
            const { userId, incidentId } = req.params;
            
            const { data: videos, error } = await supabase
                .from('incident_evidence')
                .select('*')
                .eq('user_id', userId)
                .eq('incident_id', incidentId)
                .eq('kind', 'video')
                .eq('source', 'upload')
                .order('created_at', { ascending: false });
            
            if (error) {
                throw error;
            }
            
            res.json({
                success: true,
                videos: videos || [],
                count: videos?.length || 0,
                requestId: req.requestId
            });
            
        } catch (error) {
            console.error('Error fetching user videos:', error);
            res.status(500).json({
                error: 'Failed to fetch videos',
                details: error.message,
                requestId: req.requestId
            });
        }
    });
    
    // Delete a video
    app.delete('/api/dashcam/video/:evidenceId', checkSharedKey, async (req, res) => {
        try {
            const { evidenceId } = req.params;
            const { userId } = req.body; // User ID should be provided in request body
            
            if (!userId) {
                return res.status(400).json({
                    error: 'User ID required',
                    requestId: req.requestId
                });
            }
            
            // Get the evidence record
            const { data: evidence, error: fetchError } = await supabase
                .from('incident_evidence')
                .select('storage_path, user_id')
                .eq('id', evidenceId)
                .single();
            
            if (fetchError || !evidence) {
                return res.status(404).json({
                    error: 'Video not found',
                    requestId: req.requestId
                });
            }
            
            // Verify user owns this video
            if (evidence.user_id !== userId) {
                return res.status(403).json({
                    error: 'Access denied',
                    requestId: req.requestId
                });
            }
            
            // Delete from storage
            const { error: storageError } = await supabase.storage
                .from('dash-cam-upload')
                .remove([evidence.storage_path]);
            
            if (storageError) {
                console.warn('Failed to delete from storage:', storageError);
                // Continue with database deletion
            }
            
            // Delete database record
            const { error: dbError } = await supabase
                .from('incident_evidence')
                .delete()
                .eq('id', evidenceId);
            
            if (dbError) {
                throw dbError;
            }
            
            res.json({
                success: true,
                message: 'Video deleted successfully',
                requestId: req.requestId
            });
            
        } catch (error) {
            console.error('Error deleting video:', error);
            res.status(500).json({
                error: 'Failed to delete video',
                details: error.message,
                requestId: req.requestId
            });
        }
    });
    
    console.log('🎥 Dash-cam API endpoints registered');
}
