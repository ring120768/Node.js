4. New API Endpoints
Add these new endpoints to your index.js:
javascript// ============================================
// SUMMARY PREVIEW ENDPOINTS
// ============================================

// Get summary preview
app.get('/api/summary/preview/:queueId', async (req, res) => {
    try {
        const { queueId } = req.params;
        const userId = req.headers['x-user-id'] || req.query.userId;

        // Verify ownership
        const { data: queueItem, error } = await supabase
            .from('transcription_queue')
            .select('*')
            .eq('id', queueId)
            .eq('user_id', userId)
            .single();

        if (error || !queueItem) {
            return res.status(404).json({ 
                error: 'Summary not found or access denied' 
            });
        }

        if (queueItem.status !== TRANSCRIPTION_STATUS.PENDING_REVIEW) {
            return res.status(400).json({ 
                error: 'Summary is not in preview status' 
            });
        }

        // Mark as viewed
        if (!queueItem.preview_viewed_at) {
            await supabase
                .from('transcription_queue')
                .update({ preview_viewed_at: new Date().toISOString() })
                .eq('id', queueId);
        }

        // Get related incident report data for context
        const { data: incidentData } = await supabase
            .from('incident_reports')
            .select('accident_date, location, description')
            .eq('user_id', userId)
            .order('created_at', { ascending: false })
            .limit(1)
            .single();

        res.json({
            queueId,
            summary: queueItem.ai_summary,
            transcription: queueItem.transcription_text,
            metadata: {
                created_at: queueItem.created_at,
                audio_file_url: queueItem.audio_file_url,
                incident_data: incidentData
            },
            editable: true,
            status: queueItem.status
        });

        // Log GDPR activity
        await GDPR.logActivity(
            userId,
            'SUMMARY_PREVIEW_VIEWED',
            'transcription_queue',
            { queue_id: queueId }
        );

    } catch (error) {
        Logger.error('Error fetching preview:', error);
        res.status(500).json({ error: 'Failed to fetch preview' });
    }
});

// Save edited summary
app.post('/api/summary/save/:queueId', async (req, res) => {
    try {
        const { queueId } = req.params;
        const { summary, skipPreview } = req.body;
        const userId = req.headers['x-user-id'] || req.body.userId;

        // Verify ownership
        const { data: queueItem, error: fetchError } = await supabase
            .from('transcription_queue')
            .select('*')
            .eq('id', queueId)
            .eq('user_id', userId)
            .single();

        if (fetchError || !queueItem) {
            return res.status(404).json({ 
                error: 'Summary not found or access denied' 
            });
        }

        // Check if summary was edited
        const wasEdited = summary !== queueItem.original_summary;

        // Update the summary
        const { error: updateError } = await supabase
            .from('transcription_queue')
            .update({
                ai_summary: summary,
                status: TRANSCRIPTION_STATUS.COMPLETED,
                summary_edited: wasEdited,
                edited_by: wasEdited ? userId : null,
                completed_at: new Date().toISOString()
            })
            .eq('id', queueId);

        if (updateError) throw updateError;

        // Send WebSocket notification
        broadcastToUser(userId, {
            type: 'SUMMARY_SAVED',
            queueId,
            message: 'Summary has been saved successfully'
        });

        // Log GDPR activity
        await GDPR.logActivity(
            userId,
            wasEdited ? 'SUMMARY_EDITED_AND_SAVED' : 'SUMMARY_APPROVED',
            'transcription_queue',
            { 
                queue_id: queueId,
                was_edited: wasEdited
            }
        );

        res.json({ 
            success: true, 
            message: 'Summary saved successfully',
            wasEdited 
        });

    } catch (error) {
        Logger.error('Error saving summary:', error);
        res.status(500).json({ error: 'Failed to save summary' });
    }
});

// Skip preview and auto-approve
app.post('/api/summary/skip-preview/:queueId', async (req, res) => {
    try {
        const { queueId } = req.params;
        const userId = req.headers['x-user-id'] || req.body.userId;

        // Verify ownership and status
        const { data: queueItem, error } = await supabase
            .from('transcription_queue')
            .select('*')
            .eq('id', queueId)
            .eq('user_id', userId)
            .eq('status', TRANSCRIPTION_STATUS.PENDING_REVIEW)
            .single();

        if (error || !queueItem) {
            return res.status(404).json({ 
                error: 'Summary not found or not in preview status' 
            });
        }

        // Update status to completed
        await supabase
            .from('transcription_queue')
            .update({
                status: TRANSCRIPTION_STATUS.COMPLETED,
                preview_required: false,
                completed_at: new Date().toISOString()
            })
            .eq('id', queueId);

        // Send notification
        broadcastToUser(userId, {
            type: 'SUMMARY_AUTO_APPROVED',
            queueId,
            message: 'Summary has been auto-approved'
        });

        res.json({ success: true });

    } catch (error) {
        Logger.error('Error skipping preview:', error);
        res.status(500).json({ error: 'Failed to skip preview' });
    }
});

// Update user preview preference
app.post('/api/user/preview-preference', async (req, res) => {
    try {
        const { enabled } = req.body;
        const userId = req.headers['x-user-id'] || req.body.userId;

        // Upsert user preference
        const { error } = await supabase
            .from('user_preferences')
            .upsert({
                user_id: userId,
                preview_enabled: enabled,
                updated_at: new Date().toISOString()
            })
            .eq('user_id', userId);

        if (error) throw error;

        res.json({ 
            success: true, 
            preview_enabled: enabled 
        });

    } catch (error) {
        Logger.error('Error updating preview preference:', error);
        res.status(500).json({ error: 'Failed to update preference' });
    }
});
