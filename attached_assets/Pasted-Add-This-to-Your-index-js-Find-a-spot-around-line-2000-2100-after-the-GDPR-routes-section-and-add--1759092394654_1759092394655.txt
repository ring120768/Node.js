Find a spot around line 2000-2100 (after the GDPR routes section) and add:
javascript// Simple webhook endpoint for testing
app.post('/webhook/signup-simple', checkSharedKey, async (req, res) => {
  try {
    console.log('Simple webhook received:', {
      userId: req.body.create_user_id,
      timestamp: new Date().toISOString()
    });
    
    // Basic validation
    if (!req.body.create_user_id) {
      return res.status(400).json({
        error: 'Missing create_user_id',
        requestId: req.requestId
      });
    }
    
    // For testing - just save to database if configured
    if (supabaseEnabled) {
      const { data, error } = await supabase
        .from('user_signup')
        .insert({
          create_user_id: req.body.create_user_id,
          full_name: req.body.full_name || 'Test User',
          email: req.body.email || 'test@example.com',
          created_at: new Date().toISOString()
        })
        .select()
        .single();
      
      if (error) {
        console.error('Database error:', error);
        return res.status(500).json({
          error: 'Database error',
          details: error.message,
          requestId: req.requestId
        });
      }
      
      res.json({
        success: true,
        message: 'Test webhook processed',
        data: data,
        requestId: req.requestId
      });
    } else {
      res.status(503).json({
        error: 'Supabase not configured',
        requestId: req.requestId
      });
    }
    
  } catch (error) {
    console.error('Simple webhook error:', error);
    res.status(500).json({
      error: 'Internal server error',
      message: error.message,
      requestId: req.requestId
    });
  }
});
