app.post('/webhook/signup', checkSharedKey, async (req, res) => {
  try {
    Logger.info('Webhook signup received', { 
      userId: req.body.create_user_id,
      email: req.body.email 
    });

    // Validate required fields
    if (!req.body.create_user_id || !req.body.email) {
      return res.status(400).json({
        error: 'Missing required fields',
        required: ['create_user_id', 'email'],
        requestId: req.requestId
      });
    }

    // Check if Supabase is configured
    if (!supabaseEnabled) {
      Logger.warn('Webhook received but Supabase not configured');
      return res.status(503).json({
        error: 'Database service not configured',
        message: 'Please configure Supabase connection',
        requestId: req.requestId
      });
    }

    // Process the signup data
    const signupData = {
      create_user_id: req.body.create_user_id,
      full_name: req.body.full_name || '',
      email: req.body.email,
      phone_number: req.body.phone_number || '',
      incident_date: req.body.incident_date || null,
      incident_location: req.body.incident_location || '',
      detailed_account_of_what_happened: req.body.detailed_account_of_what_happened || '',
      gdpr_consent: req.body.gdpr_consent || false,
      legal_support: req.body.legal_support || '',
      created_at: new Date().toISOString()
    };

    // Save to database
    const { data, error } = await supabase
      .from('user_signup')
      .upsert(signupData, {
        onConflict: 'create_user_id'
      })
      .select()
      .single();

    if (error) {
      Logger.error('Database error saving signup:', error);
      return res.status(500).json({
        error: 'Failed to save signup data',
        details: error.message,
        requestId: req.requestId
      });
    }

    // Log GDPR consent if module is available
    if (gdprModule && signupData.gdpr_consent) {
      await gdprModule.recordConsent(
        signupData.create_user_id,
        'marketing',
        true,
        'webhook_signup'
      );
    }

    Logger.success('Webhook signup processed successfully', {
      userId: signupData.create_user_id
    });

    res.json({
      success: true,
      message: 'Signup data received and stored',
      userId: signupData.create_user_id,
      requestId: req.requestId
    });

  } catch (error) {
    Logger.error('Webhook signup error:', error);
    res.status(500).json({
      error: 'Internal server error',
      message: error.message,
      requestId: req.requestId
    });
  }
});
